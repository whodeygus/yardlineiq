    <script>
        // Simple admin authentication
        const adminPassword = prompt('Enter admin password:');
        if (adminPassword !== 'YardLineIQ2024!') {
            alert('Access denied');
            window.location.href = '/';
        }

        async function loadStats() {
            try {
                const response = await fetch('/api/admin/stats');
                const stats = await response.json();
                
                if (response.ok) {
                    document.getElementById('totalUsers').textContent = stats.totalUsers || 0;
                    document.getElementById('paidUsers').textContent = stats.paidUsers || 0;
                    document.getElementById('totalPicks').textContent = stats.totalPicks || 0;
                    document.getElementById('winRate').textContent = stats.winRate || '61%';
                }
            } catch (error) {
                console.error('Error loading stats:', error);
            }
        }

        document.getElementById('pickForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            
            const formData = new FormData(e.target);
            const pickData = Object.fromEntries(formData);
            
            const successMsg = document.getElementById('pickSuccess');
            const errorMsg = document.getElementById('pickError');
            
            successMsg.style.display = 'none';
            errorMsg.style.display = 'none';
            
            try {
                const response = await fetch('/api/admin/create-pick', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(pickData)
                });
                
                const result = await response.json();
                
                if (response.ok) {
                    successMsg.textContent = 'Pick posted successfully!';
                    successMsg.style.display = 'block';
                    e.target.reset();
                } else {
                    errorMsg.textContent = result.error || 'Failed to post pick';
                    errorMsg.style.display = 'block';
                }
            } catch (error) {
                errorMsg.textContent = 'Network error. Please try again.';
                errorMsg.style.display = 'block';
            }
        });

        async function loadUsers() {
            try {
                const response = await fetch('/api/admin/users');
                const users = await response.json();
                
                const userList = document.getElementById('userList');
                
                if (response.ok && users.length > 0) {
                    userList.innerHTML = users.map(user => `
                        <div class="user-item">
                            <div class="user-info">
                                <div class="user-email">${user.email}</div>
                                <div class="user-subscription">${user.subscription} - Joined ${new Date(user.createdAt).toLocaleDateString()}</div>
                            </div>
                        </div>
                    `).join('');
                } else {
                    userList.innerHTML = '<div style="text-align: center; padding: 2rem; color: rgba(255,255,255,0.5);">No users found</div>';
                }
            } catch (error) {
                document.getElementById('userList').innerHTML = '<div style="text-align: center; padding: 2rem; color: #ff6b6b;">Error loading users</div>';
            }
        }

        function exportUsers() {
            alert('Export functionality would generate a CSV file with user data');
        }

        loadStats();
    </script>
